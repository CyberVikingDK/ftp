import random

tutulan_sayi = random.randint(1, 100)
print("1 ile 100 arasında bir sayı tuttum. 3 tahmin hakkın var.")

for degisken in range(3):
    tahmin = int(input(f"{degisken + 1}. tahminini gir (1-100): "))

    if tahmin == tutulan_sayi:
      print("Doğru tahmin ettin, kazandın.")
      break
    else:
      fark = abs(tutulan_sayi - tahmin)
      if fark <= 10:
        print("Çok yakın! Ama hala yanlış. Tekrar dene.")
      elif fark <= 20:
        print("Yakın, ama doğru değil. Tekrar dene.")
      else:
        print("Uzak, Tekrar dene.")

else:
    print(f"Üzgünüm, kaybettin. Tutulan sayı: {tutulan_sayi}")
